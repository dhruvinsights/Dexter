## Goal

Allow users to export three report types as downloadable PDFs:

1. **Full Technical Report** — all metrics, charts, methodology notes
2. **Executive Summary** — 1-page briefing for leadership
3. **Policy Brief** — 2-page stakeholder document

These reports are entirely backend-generated — no browser printing or puppeteer. We use **ReportLab** (Python) to build PDFs server-side.

---

## Report Specifications

### Report 1: Full Technical Report

- Cover page: title, scenario name, date, Orbital Sentinel branding
- Section 1 — Executive Summary (AI-generated, 3 paragraphs)
- Section 2 — Scenario Configuration (parameters table)
- Section 3 — Sustainability Score breakdown (component scores, grade)
- Section 4 — Metrics Summary (5 core metrics with values and context)
- Section 5 — 30-Year Collision Frequency chart (embedded as image)
- Section 6 — 30-Year Debris Growth chart (embedded as image)
- Section 7 — Orbital Shell Analysis (per-shell table)
- Section 8 — Comparative Analysis (vs baseline, side-by-side table)
- Section 9 — AI Risk Assessment (full text from WS5)
- Section 10 — AI Policy Recommendation (full text from WS5)
- Footer: methodology note, data sources, generation timestamp

**Length:** 8–12 pages

---

### Report 2: Executive Summary

- 1 page
- Headline: scenario name + overall grade
- 3-paragraph AI narrative (from WS5 executive summary)
- 3 key metrics pulled as callout boxes
- Single recommendation callout
- Footer: "Generated by Orbital Sentinel · [date]"

---

### Report 3: Policy Brief

- 2 pages
- Page 1: Problem statement, current orbital environment, scenario description
- Page 2: Findings, recommendation, cost-benefit summary
- No raw data tables — narrative only with callout statistics

```python
from reportlab.lib.pagesizes import A4
from reportlab.lib.units import mm
from reportlab.lib import colors
from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
from reportlab.platypus import (
    SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle,
    HRFlowable, PageBreak
)
from reportlab.lib.enums import TA_LEFT, TA_CENTER, TA_RIGHT
from io import BytesIO
from datetime import datetime


# ── Design tokens (dark theme converted to print-friendly palette) ──
C_DARK    = colors.HexColor("#0a0e1a")
C_ACCENT  = colors.HexColor("#00d4ff")
C_TEXT    = colors.HexColor("#1e293b")
C_MUTED   = colors.HexColor("#64748b")
C_SURFACE = colors.HexColor("#f8fafc")
C_BORDER  = colors.HexColor("#e2e8f0")
C_GREEN   = colors.HexColor("#10b981")
C_RED     = colors.HexColor("#ef4444")
C_AMBER   = colors.HexColor("#f59e0b")

PAGE_WIDTH, PAGE_HEIGHT = A4
MARGIN = 20 * mm


def _styles() -> dict:
    """Build a style dictionary for the report."""
    base = getSampleStyleSheet()
    return {
        "title": ParagraphStyle(
            "title", parent=base["Normal"],
            fontSize=24, fontName="Helvetica-Bold",
            textColor=C_DARK, spaceAfter=4,
        ),
        "subtitle": ParagraphStyle(
            "subtitle", parent=base["Normal"],
            fontSize=11, fontName="Helvetica",
            textColor=C_MUTED, spaceAfter=12,
        ),
        "h2": ParagraphStyle(
            "h2", parent=base["Normal"],
            fontSize=14, fontName="Helvetica-Bold",
            textColor=C_DARK, spaceBefore=16, spaceAfter=8,
            borderPadding=(0, 0, 4, 0),
        ),
        "h3": ParagraphStyle(
            "h3", parent=base["Normal"],
            fontSize=11, fontName="Helvetica-Bold",
            textColor=C_TEXT, spaceBefore=10, spaceAfter=4,
        ),
        "body": ParagraphStyle(
            "body", parent=base["Normal"],
            fontSize=9.5, fontName="Helvetica",
            textColor=C_TEXT, leading=16, spaceAfter=6,
        ),
        "mono": ParagraphStyle(
            "mono", parent=base["Normal"],
            fontSize=9, fontName="Courier",
            textColor=C_TEXT, leading=14,
        ),
        "caption": ParagraphStyle(
            "caption", parent=base["Normal"],
            fontSize=8, fontName="Helvetica",
            textColor=C_MUTED, alignment=TA_CENTER,
        ),
    }


def _metrics_table(metrics: dict, score: dict) -> Table:
    """Build the core metrics summary table."""
    s = _styles()
    grade_colors = {"A": C_GREEN, "B": C_GREEN, "C": C_AMBER, "D": C_RED, "F": C_RED}
    grade = score.get("grade", "—")

    data = [
        ["Metric", "Value", "vs Baseline", "Status"],
        ["Collision Frequency", f"{metrics['collision_frequency']:.1f}/yr",
         f"{metrics.get('collision_reduction_pct', 0):+.1f}%", ""],
        ["Debris Growth (30yr)", f"{metrics['debris_growth_pct']:+.1f}%",
         f"{metrics.get('debris_reduction_pct', 0):+.1f}%", ""],
        ["Survivability Index", f"{metrics['survivability_pct']:.1f}%", "—", ""],
        ["Congestion Index", f"{metrics['congestion_index']:.0f}", "—", ""],
        ["Sustainability Score", f"{score.get('total_score', 0):.0f}/100",
         f"Grade: {grade}", ""],
    ]

    table = Table(data, colWidths=[60*mm, 40*mm, 40*mm, 30*mm])
    table.setStyle(TableStyle([
        # Header row
        ("BACKGROUND", (0,0), (-1,0), C_DARK),
        ("TEXTCOLOR",  (0,0), (-1,0), colors.white),
        ("FONTNAME",   (0,0), (-1,0), "Helvetica-Bold"),
        ("FONTSIZE",   (0,0), (-1,0), 9),
        ("ALIGN",      (0,0), (-1,-1), "LEFT"),
        ("ALIGN",      (1,0), (-1,-1), "CENTER"),
        # Body rows
        ("FONTSIZE",   (0,1), (-1,-1), 9),
        ("FONTNAME",   (0,1), (-1,-1), "Helvetica"),
        ("ROWBACKGROUNDS", (0,1), (-1,-1), [C_SURFACE, colors.white]),
        # Grid
        ("GRID",       (0,0), (-1,-1), 0.5, C_BORDER),
        ("TOPPADDING", (0,0), (-1,-1), 5),
        ("BOTTOMPADDING", (0,0), (-1,-1), 5),
        ("LEFTPADDING",   (0,0), (-1,-1), 6),
    ]))
    return table


def _section_header(title: str, styles: dict):
    """Returns a styled section header with underline."""
    return [
        Paragraph(title, styles["h2"]),
        HRFlowable(width="100%", thickness=1, color=C_ACCENT, spaceAfter=8),
    ]


def build_full_technical_report(
    scenario_name: str,
    config: dict,
    metrics: dict,
    score: dict,
    ai_texts: dict,  # {"executive_summary": "...", "risk_assessment": "...", "recommendation": "..."}
    shell_breakdown: list[dict],
) -> bytes:
    """
    Build the Full Technical Report PDF.
    Returns raw bytes — write to file or stream directly.
    """
    buf = BytesIO()
    doc = SimpleDocTemplate(
        buf, pagesize=A4,
        leftMargin=MARGIN, rightMargin=MARGIN,
        topMargin=MARGIN, bottomMargin=MARGIN,
    )
    styles = _styles()
    story = []

    # ── Cover Page ──
    story += [
        Spacer(1, 30*mm),
        Paragraph("ORBITAL SENTINEL", ParagraphStyle(
            "cover_title", fontSize=28, fontName="Helvetica-Bold",
            textColor=C_ACCENT, alignment=TA_CENTER
        )),
        Paragraph("Orbital Sustainability Technical Report", ParagraphStyle(
            "cover_sub", fontSize=14, fontName="Helvetica",
            textColor=C_MUTED, alignment=TA_CENTER, spaceAfter=8
        )),
        Spacer(1, 10*mm),
        HRFlowable(width="60%", thickness=2, color=C_ACCENT),
        Spacer(1, 8*mm),
        Paragraph(scenario_name, ParagraphStyle(
            "cover_scenario", fontSize=18, fontName="Helvetica-Bold",
            textColor=C_DARK, alignment=TA_CENTER
        )),
        Paragraph(f"Generated: {datetime.utcnow().strftime('%Y-%m-%d %H:%M UTC')}",
                  ParagraphStyle("cover_date", fontSize=9, fontName="Helvetica",
                                 textColor=C_MUTED, alignment=TA_CENTER)),
        PageBreak(),
    ]

    # ── Section 1: Executive Summary ──
    story += _section_header("1. Executive Summary", styles)
    story += [
        Paragraph(ai_texts.get("executive_summary", "Not available."), styles["body"]),
        Spacer(1, 8*mm),
    ]

    # ── Section 2: Sustainability Score ──
    story += _section_header("2. Sustainability Score", styles)
    grade = score.get("grade", "—")
    story += [
        Paragraph(f"Overall Score: <b>{score.get('total_score', 0):.0f}/100</b>  ·  Grade: <b>{grade}</b>",
                  styles["body"]),
        Spacer(1, 4*mm),
    ]
    # Component scores table
    comp_data = [["Component", "Score", "Weight"]] + [
        [k.replace("_", " ").title(), f"{v:.0f}/100",
         f"{score.get('weights', {}).get(k, 0)*100:.0f}%"]
        for k, v in (score.get("component_scores") or {}).items()
    ]
    comp_table = Table(comp_data, colWidths=[80*mm, 40*mm, 40*mm])
    comp_table.setStyle(TableStyle([
        ("BACKGROUND", (0,0), (-1,0), C_DARK),
        ("TEXTCOLOR",  (0,0), (-1,0), colors.white),
        ("FONTNAME",   (0,0), (-1,0), "Helvetica-Bold"),
        ("FONTSIZE",   (0,0), (-1,-1), 9),
        ("ROWBACKGROUNDS", (0,1), (-1,-1), [C_SURFACE, colors.white]),
        ("GRID", (0,0), (-1,-1), 0.5, C_BORDER),
        ("ALIGN", (1,0), (-1,-1), "CENTER"),
        ("TOPPADDING", (0,0), (-1,-1), 5),
        ("BOTTOMPADDING", (0,0), (-1,-1), 5),
        ("LEFTPADDING", (0,0), (-1,-1), 6),
    ]))
    story += [comp_table, Spacer(1, 8*mm)]

    # ── Section 3: Simulation Metrics ──
    story += _section_header("3. Simulation Metrics", styles)
    story += [_metrics_table(metrics, score), Spacer(1, 8*mm)]

    # ── Section 4: Shell-Level Analysis ──
    story += _section_header("4. Orbital Shell Analysis", styles)
    shell_data = [["Shell", "Initial Debris", "Final Debris", "Change", "Total Collisions"]]
    for s in shell_breakdown:
        change = s.get("debris_change_pct", 0)
        shell_data.append([
            s["shell_label"],
            f"{s['initial_debris']:.0f}",
            f"{s['final_debris']:.0f}",
            f"{change:+.1f}%",
            f"{s['total_collisions']:.1f}",
        ])
    shell_table = Table(shell_data, colWidths=[35*mm, 30*mm, 30*mm, 25*mm, 35*mm])
    shell_table.setStyle(TableStyle([
        ("BACKGROUND", (0,0), (-1,0), C_DARK),
        ("TEXTCOLOR",  (0,0), (-1,0), colors.white),
        ("FONTNAME",   (0,0), (-1,0), "Helvetica-Bold"),
        ("FONTSIZE",   (0,0), (-1,-1), 8.5),
        ("ROWBACKGROUNDS", (0,1), (-1,-1), [C_SURFACE, colors.white]),
        ("GRID", (0,0), (-1,-1), 0.5, C_BORDER),
        ("ALIGN", (1,0), (-1,-1), "RIGHT"),
        ("TOPPADDING", (0,0), (-1,-1), 4),
        ("BOTTOMPADDING", (0,0), (-1,-1), 4),
        ("LEFTPADDING", (0,0), (-1,-1), 5),
    ]))
    story += [shell_table, Spacer(1, 8*mm)]

    # ── Section 5: AI Risk Assessment ──
    story += _section_header("5. Risk Assessment", styles)
    story += [Paragraph(ai_texts.get("risk_assessment", "Not available."), styles["body"]),
              Spacer(1, 8*mm)]

    # ── Section 6: Policy Recommendation ──
    story += _section_header("6. Policy Recommendation", styles)
    story += [Paragraph(ai_texts.get("recommendation", "Not available."), styles["body"]),
              Spacer(1, 8*mm)]

    # ── Footer note ──
    story += [
        HRFlowable(width="100%", thickness=0.5, color=C_BORDER),
        Spacer(1, 4*mm),
        Paragraph(
            "Methodology: Simulation conducted using MIT MOCAT (Monte Carlo Object Collision "
            "Assessment Tool). Orbital data sourced from CelesTrak TLE catalogue. "
            "AI analysis generated by IBM Granite-13b-instruct-v2 via IBM watsonx.ai. "
            "This report is generated for decision support purposes.",
            styles["caption"]
        ),
    ]

    doc.build(story)
    return buf.getvalue()
```

```python
# backend/app/reports/report_service.py

import uuid
from pathlib import Path
from datetime import datetime
from app.models.reports import ReportRequest, ReportType
from app.reports.pdf_builder import build_full_technical_report
from app.simulation.precomputed import load_precomputed, DEMO_SCENARIOS
from app.metrics.extractor import extract_metrics
from app.metrics.scoring import compute_sustainability_score
from app.metrics.comparator import COST_ESTIMATES
from app.ai.analyst import run_analysis
from app.models.ai import AnalysisRequest, AnalysisType

REPORTS_DIR = Path("data/reports")

async def generate_report(request: ReportRequest) -> str:
    """
    Generate a PDF report. Returns the report ID (use to download).
    """
    REPORTS_DIR.mkdir(parents=True, exist_ok=True)

    output = load_precomputed(request.scenario_id)
    baseline = load_precomputed("baseline_2024")
    config = next((s for s in DEMO_SCENARIOS if s.scenario_id == request.scenario_id), None)

    if not output:
        raise ValueError(f"Scenario {request.scenario_id} not found")

    # Get metrics and score
    metrics = extract_metrics(output, baseline)
    cost_score = COST_ESTIMATES.get(config.intervention.value if config else "baseline", 50.0)
    score = compute_sustainability_score(metrics, cost_score)

    metrics_dict = metrics.model_dump()
    metrics_dict["score"] = score.total_score
    metrics_dict["grade"] = score.grade

    # Get AI texts
    async def get_ai(analysis_type: AnalysisType, **kwargs):
        try:
            r = await run_analysis(AnalysisRequest(
                analysis_type=analysis_type,
                metrics=metrics_dict,
                scenario_name=config.name if config else request.scenario_id,
                **kwargs
            ))
            return r.content
        except Exception as e:
            return f"AI analysis unavailable: {e}"

    exec_summary = await get_ai(AnalysisType.EXECUTIVE_SUMMARY)
    risk_assessment = await get_ai(AnalysisType.RISK_ASSESSMENT)
    recommendation = await get_ai(AnalysisType.RECOMMENDATION)

    # Build PDF
    pdf_bytes = build_full_technical_report(
        scenario_name=config.name if config else request.scenario_id,
        config=config.model_dump() if config else {},
        metrics=metrics_dict,
        score=score.model_dump(),
        ai_texts={
            "executive_summary": exec_summary,
            "risk_assessment": risk_assessment,
            "recommendation": recommendation,
        },
        shell_breakdown=[s.model_dump() for s in metrics.shell_breakdown],
    )

    # Save to disk
    report_id = str(uuid.uuid4())
    report_path = REPORTS_DIR / f"{report_id}.pdf"
    report_path.write_bytes(pdf_bytes)

    return report_id
```

```python
from pydantic import BaseModel
from enum import Enum

class ReportType(str, Enum):
    FULL_TECHNICAL    = "full_technical"
    EXECUTIVE_SUMMARY = "executive_summary"
    POLICY_BRIEF      = "policy_brief"

class ReportRequest(BaseModel):
    scenario_id: str

    report_id: str
    download_url: str
    report_type: ReportType
    scenario_id: str
    generated_at: str
```

### 7.4 API Endpoints

**File:** `backend/app/routers/reports.py`

```python
from fastapi import APIRouter
from fastapi.responses import FileResponse
from pathlib import Path
from app.models.reports import ReportRequest, ReportResponse
from app.reports.report_service import generate_report
from datetime import datetime

router = APIRouter(prefix="/api/reports", tags=["reports"])

@router.post("/generate", response_model=ReportResponse)
async def create_report(request: ReportRequest):
    """
    Generate a PDF report for a scenario.
    Returns download URL immediately (report generated synchronously).
    """
    report_id = await generate_report(request)
    return ReportResponse(
        report_id=report_id,
        download_url=f"/api/reports/{report_id}/download",
        report_type=request.report_type,
        scenario_id=request.scenario_id,
        generated_at=datetime.utcnow().isoformat(),
    )

@router.get("/{report_id}/download")
async def download_report(report_id: str):
    """Stream the generated PDF to the client."""
    path = Path(f"data/reports/{report_id}.pdf")
    if not path.exists():
        from fastapi import HTTPException
        raise HTTPException(404, "Report not found")
    return FileResponse(
        path=str(path),
        media_type="application/pdf",
        filename=f"orbital-sentinel-report-{report_id[:8]}.pdf",
    )
```
